function submitAttendance() {
    const inputField = document.getElementById('attendanceInput');
    const msgEl = document.getElementById('attendanceMsg');
    
    if(!inputField) return;
    
    // `.trim()` digunakan untuk menghapus spasi kosong di awal/akhir input tanpa sengaja
    const barcodeValue = inputField.value.trim();

    if(!barcodeValue) return;

    // 1. Nilai default jika data siswa tidak ditemukan
    let namaSiswa = "Pengunjung Umum";
    let kelasSiswa = "-";

    // 2. Pencarian cerdas ke database anggota (localStudentsArray)
    if (typeof localStudentsArray !== 'undefined' && localStudentsArray.length > 0) {
        
        const siswaDitemukan = localStudentsArray.find(siswa => {
            // Kita ubah dua-duanya menjadi String murni, lalu hilangkan spasinya agar pencocokan 100% akurat
            const barcodeDatabase = siswa.barcode ? String(siswa.barcode).trim() : "";
            const barcodeInput = String(barcodeValue).trim();
            
            return barcodeDatabase === barcodeInput;
        });
        
        // Jika ditemukan, ambil data asli dari database siswa Anda
        if(siswaDitemukan) {
            namaSiswa = siswaDitemukan.name || siswaDitemukan.nama || "Siswa Aktif";
            kelasSiswa = siswaDitemukan.class || siswaDitemukan.kelas || "-";
        }
    }

    const now = new Date();
    const timeString = now.toLocaleTimeString('id-ID', { hour: '2-digit', minute: '2-digit', second: '2-digit' }) + " WIB";
    const dateString = now.toLocaleDateString('id-ID');

    const newLog = {
        waktu: timeString,
        tanggal: dateString,
        id: barcodeValue,
        nama: namaSiswa,
        status: kelasSiswa
    };

    // Masukkan ke baris paling atas di tabel log absensi
    attendanceData.unshift(newLog);
    localStorage.setItem('attendance_logs', JSON.stringify(attendanceData));

    // Memunculkan pesan sukses di bawah kotak input
    msgEl.style.color = '#2ecc71';
    msgEl.innerHTML = `<i class="fa-solid fa-circle-check"></i> Absen Berhasil! Selamat Datang, <b>${namaSiswa}</b> (${kelasSiswa})`;
    msgEl.style.display = 'block';

    // Reset kotak input agar siap discan kembali
    inputField.value = '';
    inputField.focus();
    renderAttendanceTable();
    calculateAttendanceStats();

    // Sembunyikan notifikasi setelah 4 detik
    setTimeout(() => { msgEl.style.display = 'none'; }, 4000);
}
function renderAttendanceTable() {
    const tbody = document.getElementById('attendanceTableBody');
    if(!tbody) return;

    const todayStr = new Date().toLocaleDateString('id-ID');
    const todayLogs = attendanceData.filter(item => item.tanggal === todayStr);

    if(todayLogs.length === 0) {
        tbody.innerHTML = `<tr><td colspan="5" style="text-align: center; color: var(--text-gray);">Belum ada data absensi masuk hari ini.</td></tr>`;
        return;
    }

    tbody.innerHTML = todayLogs.map((item, index) => `
        <tr>
            <td><b>${index + 1}</b></td>
            <td><i class="fa-regular fa-clock" style="color: var(--primary-blue)"></i> ${item.waktu}</td>
            <td><code>${item.id}</code></td>
            <td><strong>${item.nama}</strong></td>
            <td><span class="badge badge-returned" style="background-color: var(--primary-blue); color: white; padding: 4px 10px; border-radius: 8px; font-size: 0.85rem;">${item.status}</span></td>
        </tr>
    `).join('');
}

function calculateAttendanceStats() {
    const todayStr = new Date().toLocaleDateString('id-ID');
    
    const totalToday = attendanceData.filter(item => item.tanggal === todayStr).length;
    const txtTotal = document.getElementById('txtTotalAbsen');
    if(txtTotal) txtTotal.innerText = totalToday;

    const uniqueDates = [...new Set(attendanceData.map(item => item.tanggal))];
    const totalDaysRecorded = uniqueDates.length || 1;
    const totalAllDays = attendanceData.length;

    const averagePerDay = (totalAllDays / totalDaysRecorded).toFixed(1);
    const txtAvg = document.getElementById('txtRataRataAbsen');
    if(txtAvg) txtAvg.innerText = averagePerDay;
}

        window.processBookReturn = async function() {
            const barcodeInput = document.getElementById('returnBookBarcodeInput').value.trim();
            const successBox = document.getElementById('returnSuccessBox');
            
            if(!barcodeInput) return;

            try {
                const q = query(loansCollection, where("bookBarcode", "==", barcodeInput), where("status", "==", "Dipinjam"));
                const querySnapshot = await getDocs(q);

                if(querySnapshot.empty) {
                    alert("Gagal: Buku dengan barcode tersebut tidak ditemukan dalam daftar aktif peminjaman!");
                    document.getElementById('returnBookBarcodeInput').value = "";
                    document.getElementById('returnBookBarcodeInput').focus();
                    successBox.style.display = "none";
                    return;
                }

                querySnapshot.forEach(async (docSnap) => {
                    const loanId = docSnap.id;
                    const loanData = docSnap.data();

                    await updateDoc(doc(db, "loans", loanId), {
                        status: "Kembali"
                    });

                    document.getElementById('txtReturnBookTitle').innerText = loanData.bookTitle || "Buku Tanpa Judul";
                    document.getElementById('txtReturnStudentName').innerText = `Dikembalikan oleh: ${loanData.studentName || '-'} (Kelas ${loanData.studentClass || '-'})`;
                    successBox.style.display = "flex";
                    
                    document.getElementById('returnBookBarcodeInput').value = "";
                    document.getElementById('returnBookBarcodeInput').focus();
                });

            } catch (error) {
                console.error(error);
                alert("Sistem error saat mencoba mengembalikan buku.");
            }
        }

        window.renderLoanReportTable = function(dataArray) {
            const tbody = document.getElementById('loanReportTableBody');
            if(!tbody) return;
            tbody.innerHTML = "";

            const masterCheck = document.getElementById('checkAllLoans');
            if(masterCheck) masterCheck.checked = false;
            updateDeleteButtonState();

            if(dataArray.length === 0) {
                tbody.innerHTML = `<tr><td colspan="6" style="text-align:center; color:var(--text-gray);">Belum ada riwayat sirkulasi buku.</td></tr>`;
                return;
            }

            // Ambil tanggal hari ini (format YYYY-MM-DD)
            const today = new Date();
            today.setHours(0,0,0,0);

            dataArray.forEach((loan, index) => {
                const tr = document.createElement('tr');
                
                let statusBadge = "";
                let rowStyle = "";

                if (loan.status === "Dipinjam") {
                    // Hitung selisih hari
                    const borrowDate = new Date(loan.loanDate);
                    borrowDate.setHours(0,0,0,0);
                    
                    // Selisih dalam milidetik diubah ke hari
                    const timeDiff = today.getTime() - borrowDate.getTime();
                    const daysElapsed = Math.floor(timeDiff / (1000 * 60 * 60 * 24));

                    if (daysElapsed > 7) {
                        const lateDays = daysElapsed - 7;
                        statusBadge = `<span class="badge" style="background-color: var(--danger-red);"><i class="fa-solid fa-triangle-exclamation"></i> Terlambat ${lateDays} Hari</span>`;
                        rowStyle = `style="background-color: rgba(255, 75, 75, 0.08);"`; // Beri highlight merah tipis pada baris yang terlambat
                    } else {
                        statusBadge = `<span class="badge badge-loaned"><i class="fa-solid fa-clock"></i> Dipinjam</span>`;
                    }
                } else {
                    statusBadge = `<span class="badge badge-returned"><i class="fa-solid fa-circle-check"></i> Kembali</span>`;
                }

                tr.innerHTML = `
                    <td style="text-align: center;">
                        <input type="checkbox" class="loan-checkbox" value="${loan.id}" onchange="updateDeleteButtonState()" style="transform: scale(1.2); cursor: pointer;">
                    </td>
                    <td>${index + 1}</td>
                    <td style="white-space:nowrap; font-size:0.9rem;">${loan.loanDate || '-'}</td>
                    <td>
                        <strong>${loan.studentName || '-'}</strong><br>
                        <span style="font-size:0.8rem; color:var(--text-gray);">Barcode: ${loan.studentBarcode || '-'} (Kl: ${loan.studentClass || '-'})</span>
                    </td>
                    <td>
                        <strong>${loan.bookTitle || '-'}</strong><br>
                        <span style="font-size:0.8rem; color:var(--text-gray);">Barcode: ${loan.bookBarcode || '-'}</span>
                    </td>
                    <td>${statusBadge}</td>
                `;
                
                if(rowStyle) tr.setAttribute('style', tr.getAttribute('style') ? tr.getAttribute('style') + rowStyle.replace('style="','').replace('"','') : rowStyle.replace('style="','').replace('"',''));
                tbody.appendChild(tr);
            });
        }
